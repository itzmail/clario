**C. Update delete completion in app.rs to record stats:**

In the delete_rx drain block, where `None` sentinel is received (currently sets `is_deleting = false`), add stats tracking BEFORE `retain_unselected` removes the items:

```rust
None => {
    // Record stats before removing items
    self.config.stats.total_bytes_freed += self.pending_bytes_to_free;
    self.config.stats.last_clean_date = Some(chrono::Local::now());

    if self.mode == AppMode::AppUninstaller {
        self.config.stats.total_files_deleted += self.apps.iter().filter(|a| a.is_selected).count() as u64;
    } else {
        self.config.stats.total_files_deleted += crate::core::file_ops::FileOps::count_selected(&self.scanned_files) as u64;
    }
    self.config.save();
    self.pending_bytes_to_free = 0;

    self.is_deleting = false;
    self.show_delete_confirm = false;
    self.delete_rx = None;

    if self.mode == AppMode::AppUninstaller {
        self.apps.retain(|a| !a.is_selected);
        if self.selected_app_index >= self.apps.len() {
            self.selected_app_index = self.apps.len().saturating_sub(1);
        }
        self.app_table_state.select(Some(self.selected_app_index));
    } else {
        crate::core::file_ops::FileOps::retain_unselected(&mut self.scanned_files);
    }
    break;
}
```

pub fn count_selected(files: &[crate::models::file_info::FileInfo]) -> usize {
